<script>
document.addEventListener('DOMContentLoaded', function () {
  const forms = document.querySelectorAll('form.form-with-validation');

  forms.forEach(function(form) {
    const fileInput = form.querySelector('input[type="file"]');
    const fileListElement = form.querySelector('.file-list');
    const selectedFiles = [];

    // Клик по кнопке "Загрузить файл"
    form.querySelector('.file-upload').addEventListener('click', function () {
      fileInput.click();
    });

    // При выборе файлов
    fileInput.addEventListener('change', function (event) {
      const newFiles = Array.from(event.target.files);

      newFiles.forEach(file => {
        if (!selectedFiles.some(f => f.name === file.name && f.size === file.size)) {
          selectedFiles.push(file);
        }
      });

      updateFileList();
      fileInput.value = ''; // позволяет повторно выбрать один и тот же файл
    });

    // Обновление отображения списка файлов
    function updateFileList() {
      fileListElement.innerHTML = '';

      selectedFiles.forEach((file, index) => {
        const li = document.createElement('li');
        li.textContent = file.name;

        const deleteButton = document.createElement('button');
        deleteButton.textContent = 'Удалить';
        deleteButton.classList.add('delete-file');
        deleteButton.setAttribute('type', 'button'); // важно, чтобы не отправляло форму
        deleteButton.dataset.index = index;

        deleteButton.addEventListener('click', function () {
          selectedFiles.splice(index, 1);
          updateFileList();
        });

        li.appendChild(deleteButton);
        fileListElement.appendChild(li);
      });

      fileInput.files = createFileList(selectedFiles);
    }

    // Создание FileList из массива
    function createFileList(filesArray) {
      const dataTransfer = new DataTransfer();
      filesArray.forEach(file => dataTransfer.items.add(file));
      return dataTransfer.files;
    }

    // Добавляем временные input перед submit
    form.addEventListener('submit', function(e) {
      selectedFiles.forEach(file => {
        const tempInput = document.createElement('input');
        tempInput.type = 'file';
        tempInput.name = fileInput.name;
        tempInput.files = createFileList([file]);
        tempInput.style.display = 'none';
        form.appendChild(tempInput);
      });
    });
  });
});
</script>
